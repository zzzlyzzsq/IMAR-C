00042 //----------------------------------------------------------------------
00043 //  Basic Types:  KMcoord, KMdist, KMidx
00044 //      KMcoord and KMdist are the types used for representing
00045 //      point coordinates and distances.  They can be modified by the
00046 //      user, with some care.  It is assumed that they are both numeric
00047 //      types, and that KMdist is generally of an equal or higher type
00048 //      from KMcoord.  A variable of type KMdist should be large
00049 //      enough to store the sum of squared components of a variable
00050 //      of type KMcoord for the number of dimensions needed in the
00051 //      application.  For example, the following combinations are
00052 //      legal:
00053 //
00054 //              KMcoord KMdist
00055 //              ---------       -------------------------------
00056 //              short           short, int, long, float, double
00057 //              int             int, long, float, double
00058 //              long            long, float, double
00059 //              float           float, double
00060 //              double          double
00061 //
00062 //      It is the user's responsibility to make sure that overflow does
00063 //      not occur in distance calculation.
00064 //
00065 //      The code assumes that there is an infinite distance, KM_DIST_INF
00066 //      (as large as any legal distance).  Possible values are given below:
00067 //
00068 //          Examples:
00069 //          KMdist:             double, float, long, int, short
00070 //          KM_DIST_INF:        DBL_MAX, FLT_MAX, LONG_MAX, INT_MAX, SHRT_MAX
00071 //
00072 //      The routine that dumps a tree needs to know roughly how many
00073 //      significant digits there are in a KMcoord, so it can output
00074 //      points to full precision.  This is defined in KMcoordPrec.
00075 //      If you have ANSI C++, you should be able to use the following
00076 //      values from values.h to help in computing this:
00077 //
00078 //              KMcoord                 KMcoordBits
00079 //              --------------------    -------------------------------
00080 //              short, int, long,...    BITS(short), BITS(int), ...
00081 //              float                   FSIGNIF
00082 //              double                  DSIGNIF
00083 //
00084 //      Then KMcoordPrec = KMcoordBits/(log_2 10) where log_2 10
00085 //      is the base 2 logarithm of 10.
00086 //
00087 //      KMidx is a point index.  When the data structure is built,
00088 //      the points are given as an array.  Nearest neighbor results are
00089 //      returned as an index into this array.  To make it clearer when
00090 //      this is happening, we define the integer type KMidx.
00091 //              
00092 //----------------------------------------------------------------------
00093

00107 //----------------------------------------------------------------------
00108 //  Norms and metrics:
00109 //      KM supports any Minkowski norm for defining distance.  In
00110 //      particular, for any p >= 1, the L_p Minkowski norm defines the
00111 //      length of a d-vector (v0, v1, ..., v(d-1)) to be
00112 //
00113 //              (|v0|^p + |v1|^p + ... + |v(d-1)|^p)^(1/p),
00114 //
00115 //      (where ^ denotes exponentiation, and |.| denotes absolute
00116 //      value).  The distance between two points is defined to be
00117 //      the norm of the vector joining them.  Some common distance
00118 //      metrics include
00119 //
00120 //              Euclidean metric        p = 2
00121 //              Manhattan metric        p = 1
00122 //              Max metric              p = infinity
00123 //
00124 //      In the case of the max metric, the norm is computed by
00125 //      taking the maxima of the absolute values of the components.
00126 //      KM is highly "coordinate-based" and does not support general
00127 //      distances functions (e.g. those obeying just the triangle
00128 //      inequality).  It also does not support distance functions
00129 //      based on inner-products.
00130 //
00131 //      For the purpose of computing nearest neighbors, it is not
00132 //      necessary to compute the final power (1/p).  Thus the only
00133 //      component that is used by the program is |v(i)|^p.
00134 //
00135 //      KM parameterizes the distance computation through the following
00136 //      macros.  (Macros are used rather than procedures for efficiency.)
00137 //      Recall that the distance between two points is given by the length
00138 //      of the vector joining them, and the length or norm of a vector v
00139 //      is given by formula:
00140 //
00141 //              |v| = ROOT(POW(v0) # POW(v1) # ... # POW(v(d-1)))
00142 //
00143 //      where ROOT, POW are unary functions and # is an associative and
00144 //      commutative binary operator satisfying:
00145 //
00146 //          **  POW:    coord           --> dist
00147 //          **  #:      dist x dist     --> dist
00148 //          **  ROOT:   dist (>0)       --> double
00149 //
00150 //      For early termination in distance calculation (partial distance
00151 //      calculation) we assume that POW and # together are monotonically
00152 //      increasing on sequences of arguments, meaning that for all
00153 //      v0..vk and y:
00154 //
00155 //      POW(v0) #...# POW(vk) <= (POW(v0) #...# POW(vk)) # POW(y).
00156 //
00157 //      Due to the use of incremental distance calculations in the code
00158 //      for searching k-d trees, we assume that there is an incremental
00159 //      update function DIFF(x,y) for #, such that if:
00160 //
00161 //                  s = x0 # ... # xi # ... # xk 
00162 //
00163 //      then if s' is s with xi replaced by y, that is, 
00164 //      
00165 //                  s' = x0 # ... # y # ... # xk
00166 //
00167 //      can be computed by:
00168 //
00169 //                  s' = s # DIFF(xi,y).
00170 //
00171 //      Thus, if # is + then DIFF(xi,y) is (yi-x).  For the L_infinity
00172 //      norm we make use of the fact that in the program this function
00173 //      is only invoked when y > xi, and hence DIFF(xi,y)=y.
00174 //
00175 //      Finally, for approximate nearest neighbor queries we assume
00176 //      that POW and ROOT are related such that
00177 //
00178 //                  v*ROOT(x) = ROOT(POW(v)*x)
00179 //
00180 //      Here are the values for the various Minkowski norms:
00181 //
00182 //      L_p:    p even:                         p odd:
00183 //              -------------------------       ------------------------
00184 //              POW(v)          = v^p           POW(v)          = |v|^p
00185 //              ROOT(x)         = x^(1/p)       ROOT(x)         = x^(1/p)
00186 //              #               = +             #               = +
00187 //              DIFF(x,y)       = y - x         DIFF(x,y)       = y - x 
00188 //
00189 //      L_inf:
00190 //              POW(v)          = |v|
00191 //              ROOT(x)         = x
00192 //              #               = max
00193 //              DIFF(x,y)       = y
00194 //
00195 //      By default the Euclidean norm is assumed.  To change the norm,
00196 //      uncomment the appropriate set of macros below.
00197 //
00198 //----------------------------------------------------------------------
00199 
00200 //----------------------------------------------------------------------
00201 //      Use the following for the Euclidean norm
00202 //----------------------------------------------------------------------
00203 #define KM_POW(v)               ((v)*(v))
00204 #define KM_ROOT(x)              sqrt(x)
00205 #define KM_SUM(x,y)             ((x) + (y))
00206 #define KM_DIFF(x,y)            ((y) - (x))

00364 //  Orthogonal (axis aligned) rectangle
00365 //      Orthogonal rectangles are represented by two points, one
00366 //      for the lower left corner (min coordinates) and the other
00367 //      for the upper right corner (max coordinates).
00368 //
00369 //      The constructor initializes from either a pair of coordinates,
00370 //      pair of points, or another rectangle.  Note that all constructors
00371 //      allocate new point storage.  The destructor deallocates this
00372 //      storage.
00373 //
00374 //      BEWARE: Orthogonal rectangles should be passed ONLY BY REFERENCE.
00375 //      (C++'s default copy constructor will not allocate new point
00376 //      storage, then on return the destructor free's storage, and then
00377 //      you get into big trouble in the calling procedure.)
