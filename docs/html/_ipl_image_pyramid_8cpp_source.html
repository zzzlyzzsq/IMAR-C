00150 std::size_t IplImagePyramid::getIndex(double scaleFactor, int round) const
00151 {
00152         // make sure that we don't break for the two extreme cases:
00153         // scale factor too large and scale factor too small
00154         if (scaleFactor >= _scaleFactors[_scaleFactors.size() - 1])
00155                 return _scaleFactors.size() - 1;
00156         if (scaleFactor <= 1)
00157                 return 0;
00158 
00159         // find the correct answer based on the given rounding type
00160         std::size_t i = 0;
00161         if (round < 0) {
00162                 // find the next level with a smaller factor
00163                 scaleFactor += _epsilon;
00164                 for (i = 1; scaleFactor > _scaleFactors[i] && i < _scaleFactors.size(); ++i);
00165                 --i;
00166         }
00167         else if (round > 0) {
00168                 // find the next level with a bigger factor
00169                 scaleFactor -= _epsilon;
00170                 for (i = 0; scaleFactor >= _scaleFactors[i] && i < _scaleFactors.size() - 1; ++i);
00171         }
00172         else {
00173                 // find the closest level with respect to the scale factor
00174                 double bestDist = fabs(_scaleFactors[0] - scaleFactor);
00175                 std::size_t iBest = 0;
00176                 for (i = 1; i < _scaleFactors.size(); ++i) {
00177                         double dist = fabs(_scaleFactors[i] - scaleFactor);
00178                         if (dist < bestDist) {
00179                                 iBest = i;
00180                                 bestDist = dist;
00181                         }
00182                         else
00183                                 break;
00184                 }
00185                 i = iBest;
00186         }
00187         return i;
00188 }
