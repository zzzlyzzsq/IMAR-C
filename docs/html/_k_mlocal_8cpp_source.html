00029 KMfilterCenters KMlocal::execute()              // execute the algorithm
00030 {
00031    reset();                                     // resets everything
00032    while (!isDone()) {                          // while not done
00033         beginRun();                             // start a new run
00034         do {                                    // do while run is not done
00035             beginStage();                       // start of stage processing
00036             KMalg method = selectMethod();      // select a method
00037             switch(method) {                    // apply one stage
00038             case LLOYD:                         // Lloyd's algorithm
00039                 curr.lloyd1Stage();
00040                 break;
00041             case SWAP:                          // swap heuristic
00042                 curr.swap1Stage();
00043                 break;
00044             case RANDOM:                        // get random centers
00045                 curr.genRandom();
00046                 break;
00047             default:                            // shouldn't come here
00048                 assert(false);
00049                 break;
00050             }
00051             endStage();                         // end of stage processing
00052         } while (!isRunDone());                 // while run is not done
00053         endRun();                               // end of run processing
00054         tryAcceptance();                        // accept if appropriate
00055     }
00056     return best;                                // return best solution
00057 }
